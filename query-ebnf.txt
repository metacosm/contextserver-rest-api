StringLiteral   ::= '"' [^"]* '"' | "'" [^']* "'"
NumberLiteral   ::= [0-9]+ ( '.' [0-9]+)
FunctionName    ::= [a-zA-Z][0-9a-zA-Z]*
FunctionArg     ::= ( StringLiteral | NumberLiteral )
FunctionArgs    ::= FunctionArg? ( ',' FunctionArg )*
FunctionCall    ::= FunctionName '(' FunctionArgs ')'
Clause          ::= ( 'NOT' FunctionCall | FunctionCall 'AND' FunctionCall | FunctionCall 'OR' FunctionCall )
ParenClause     ::= '(' Clause ')'
Query           ::= Clause ( Clause | ParenClause )*
